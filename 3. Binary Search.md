### Binary Search
#### Recursive Approach
```java
class Solution {
    public int search(int[] nums, int target) {
        int n = nums.length;
        int low = 0;
        int high = n - 1;
        return binarysearch(nums, low, high, target); 
    }

    public int binarysearch(int[] nums, int low, int high, int target) {
        if (low > high) {
            return -1;
        }
        
        int mid = low + (high - low) / 2; 

        if (nums[mid] == target) {
            return mid;
        } else if (nums[mid] > target) { 
            return binarysearch(nums, low, mid - 1, target);
        } else { 
            return binarysearch(nums, mid + 1, high, target);
        }
    }
}
```
#### Iterative Approach
```java
class Solution {
    public int search(int[] nums, int target) {
        int n = nums.length;
        int low = 0;
        int high = n-1;
        
        while(low <= high){
            int mid = (low + high) / 2;
            if(nums[mid] == target){
                return mid;
            } else if (target > nums[mid]){
                low = mid + 1;
            } else {
                high = mid-1;
            }
        }
        return -1;
    }
}
```

```java
                ans = mid;
                //look for smaller index on the left
                high = mid - 1;

            }
```

```java
class Solution {
    public int[] searchRange(int[] nums, int target) {
        int n = nums.length;
        int[] result = new int[2];
        int first = -1;
        int last = -1;

        int low = 0;
        int high = n - 1;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            if (nums[mid] == target) {
                first = mid;
                high = mid - 1;
            } else if (nums[mid] < target) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        result[0] = first;

        low = 0;
        high = n - 1;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            if (nums[mid] == target) {
                last = mid;
                low = mid + 1;
            } else if (nums[mid] < target) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        result[1] = last;

        return result;
    }
}
```

```java
                queries[i] = -1;
            }
        }

    }
```

### Search in rotated Sorted Array I
```java
class Solution {
    public int search(int[] nums, int target) {
        int n = nums.length;
        int low = 0;
        int high = n - 1;

        while (low <= high) {
            int mid = low + (high - low) / 2;

            if (nums[mid] == target) {
                return mid;
            }

            if (nums[low] <= nums[mid]) {
                if (nums[low] <= target && target < nums[mid]) {
                    high = mid - 1;
                } else {
                    low = mid + 1;
                }
            } 
            else {
                if (nums[mid] < target && target <= nums[high]) {
                    low = mid + 1;
                } else {
                    high = mid - 1;
                }
            }
        }
        
        return -1;
    }
}
```

### Search in Rotated Sorted Array II
```java
class Solution {
    public boolean search(int[] nums, int target) {
        int n= nums.length;
        int low = 0;
        int high = n-1;

        while(low<=high){
            int mid = (low+high)/2;
            if(nums[mid] == target){
                return true;
            }
            if(nums[low] == nums[mid] && nums[high] == nums[mid]){
                low = low + 1;
                high = high - 1;
                continue;
            }
            if(nums[low] <= nums[mid]){
                if(nums[low] <= target && nums[mid] >= target){
                    high = mid - 1;
                } else{
                    low = mid + 1;
                }
            } else {
                if(nums[mid] <= target && nums[high] >= target){
                    low = mid + 1;
                } else {
                    high = mid - 1;
                }
            }
        }
        return false;
    }
}
```

```java
                // Eliminate left half:
                low = mid + 1;

                high = mid - 1;

        }
```

```java
    public int findMin(int[] nums) {
        int ans = Integer.MAX_VALUE;
        int n = nums.length;
        int low = 0;
        int high = n-1;
        int index = -1;

        while(low<=high){
            int mid = (low+high)/2;

            if(nums[low] <= nums[high]){
                if(nums[low] < ans){
                    ans = nums[low];
                    index = low;
                }
            }

            if(nums[low] <= nums[mid]){
                if(nums[low] < ans){
                    ans = nums[low];
                    index = low;
                }

                if(nums[mid] < ans){
                    ans = nums[mid];

                }
                high = mid-1;
            }
        }
        return ans;
```

### Search Single element in a sorted array
#### Brute Force Approach
```java
class Solution {
    public int singleNonDuplicate(int[] nums) {
        int n = nums.length;

        // If only one element, return it directly
        if (n == 1) {
            return nums[0];
        }

        // Iterate over pairs checking if they are equal
        for (int i = 0; i < n - 1; i += 2) {
            if (nums[i] != nums[i + 1]) {
                // Found the single element before the end
                return nums[i];
            }
        }

        // If no mismatch found, single element is the last one
        return nums[n - 1];
    }
}
```
#### Optimal Approach
```java
class Solution {
    public int singleNonDuplicate(int[] nums) {
        int n = nums.length;
        int low = 1;
        int high = n-2;
        if(n==1){
            return nums[0];
        }
        if(nums[0] != nums[1]){
            return nums[0];
        }
        if(nums[n-1] != nums[n-2]){
            return nums[n-1];
        }

        while(low<=high){
            int mid = (low+high)/2;
            if(nums[mid] != nums[mid+1] && nums[mid] != nums[mid-1]){
                return nums[mid];
            }
            if((mid % 2 == 1 && nums[mid] == nums[mid-1]) || (mid % 2 == 0 && nums[mid] == nums[mid+1])){
                low = mid+1;
            } else {
                high = mid - 1;
            }
        }
        return -1;
    }
}
```

### Peak element in an Array
```java
class Solution {
    public int findPeakElement(int[] nums) {
        int left = 0;
        int right = nums.length - 1;

        while (left < right) {
            int mid = left + (right - left) / 2;

            // If mid element is greater than the next, peak is on the left side
            if (nums[mid] > nums[mid + 1]) {
                right = mid;
            } else {
                // Else peak is on the right side
                left = mid + 1;
            }
        }

        // When left == right, it's the peak element's index
        return left;
    }
}
```
