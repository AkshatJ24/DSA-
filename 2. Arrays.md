# 2. Arrays

## DAY 4 : 

### 2 Sum Problem

    import java.util.HashMap;
    
    class Solution {
        public int[] twoSum(int[] nums, int target) {
            int[] arr = new int[2];
            arr[0] = arr[1] = -1;
            HashMap<Integer, Integer> map = new HashMap<>();
            
            for (int i = 0; i < nums.length; i++) {
                int num = nums[i];
                int x = target - num;
                if (map.containsKey(x)) {
                    arr[0] = map.get(x);
                    arr[1] = i;
                    return arr;
                }
                map.put(nums[i], i);
            }
            return arr;
        }
    }
