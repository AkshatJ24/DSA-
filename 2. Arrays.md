### 2 Sum Problem

    import java.util.HashMap;
    
    class Solution {
        public int[] twoSum(int[] nums, int target) {
            int[] arr = new int[2];
            arr[0] = arr[1] = -1;
            HashMap<Integer, Integer> map = new HashMap<>();
            
            for (int i = 0; i < nums.length; i++) {
                int num = nums[i];
                int x = target - num;
                if (map.containsKey(x)) {
                    arr[0] = map.get(x);
                    arr[1] = i;
                    return arr;
                }
                map.put(nums[i], i);
            }
            return arr;
        }
    }

### Sort Colours [Dutch National Flag Algorithm]

    class Solution {
        public void sortColors(int[] arr) {
            int low = 0, mid = 0;
            int high = arr.length - 1;
            while(mid <= high){
                if (arr[mid] == 0){
                    // Swap arr[low] and arr[mid]
                    int temp = arr[low];
                    arr[low] = arr[mid];
                    arr[mid] = temp;
                    low++;
                    mid++;
                }
                else if(arr[mid] == 1){
                    mid++;
                }
                else {
                    // Swap arr[mid] and arr[high]
                    int temp = arr[mid];
                    arr[mid] = arr[high];
                    arr[high] = temp;
                    high--;
                }
            }
        }
    }
