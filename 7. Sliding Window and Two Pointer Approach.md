## DAY 30
### Sliding Window Maximum
```java
import java.util.ArrayDeque;
import java.util.Deque;

class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        int n = nums.length;
        if (n == 0 || k <= 0) return new int[0];

        // If k > n, return the max of the entire array (or adjust per your spec)
        if (k > n) {
            int max = Integer.MIN_VALUE;
            for (int v : nums) max = Math.max(max, v);
            return new int[] { max };
        }

        int[] res = new int[n - k + 1];
        int ri = 0;
        Deque<Integer> dq = new ArrayDeque<>(); // stores indices

        for (int i = 0; i < n; i++) {
            // Remove indices that are out of the current window [i - k + 1, i]
            while (!dq.isEmpty() && dq.peekFirst() <= i - k) dq.pollFirst();

            // Maintain decreasing order by value in dq
            while (!dq.isEmpty() && nums[dq.peekLast()] < nums[i]) dq.pollLast();

            dq.offerLast(i);

            // Record the max once the first full window is formed
            if (i >= k - 1) res[ri++] = nums[dq.peekFirst()];
        }

        return res;
    }
}
```

## Day 32
### Length of Longest Substring without any Repeating Character
```java
class Solution {
    public int lengthOfLongestSubstring(String str) {
        if (str.length() == 0) {
            return 0;
        }
        int left = 0;
        int maxLen = 0;
        Set<Character> charSet = new HashSet<>();
        for (int right = 0; right < str.length(); right++) {
            char currentChar = str.charAt(right);
            while (charSet.contains(currentChar)) {
                charSet.remove(str.charAt(left));
                left++;
            }
            charSet.add(currentChar);
            maxLen = Math.max(maxLen, right - left + 1);
        }
        return maxLen;
    }
}
```

## Day 33
### Max Consecutive Ones III
```java
class Solution {
    public int longestOnes(int[] nums, int k) {
        int zeroCount = 0;
        int maxLength = 0;
        int left = 0;

        for (int right = 0; right < nums.length; right++) {
            if (nums[right] == 0) {
                zeroCount++;
            }
            while (zeroCount > k) {
                if (nums[left] == 0) {
                    zeroCount--;
                }
                left++;
            }
            maxLength = Math.max(maxLength, right - left + 1);
        }
        return maxLength;
    }
}
```

## Day 34
### Longest Repeating Character Replacement
```java
class Solution {
    public int characterReplacement(String s, int k) {
        int n = s.length(), l=0, maxF = Integer.MIN_VALUE, ans = 0;
        int[] freq = new int[26];
        for(int r=0;r<n;r++){
            freq[s.charAt(r) - 'A']++;
            maxF = Math.max(maxF, freq[s.charAt(r) - 'A']);
            int numC = (r-l+1) - maxF;
            if(numC > k){
                freq[s.charAt(l) - 'A']--;
                l++;
            }
            if(numC <= k){
                ans = Math.max(r-l+1, ans);
            }
        }
        return ans;
    }
}
```

## Day 35
### Binary Subarrays with Sum
```java
class Solution {
    public int numSubarraysWithSum(int[] nums, int goal) {
        return atMost(nums, goal) - atMost(nums, goal - 1);
    }
    
    private int atMost(int[] nums, int goal) {
        if (goal < 0) {
            return 0;
        }

        int count = 0;
        int currentSum = 0;
        int left = 0;

        for (int right = 0; right < nums.length; right++) {
            currentSum += nums[right];
            
            while (currentSum > goal) {
                currentSum -= nums[left];
                left++;
            }

            count += (right - left + 1);
        }
        return count;
    }
}
```

## DAY 36
### Number of Nice Subarrays
```java
class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int left = 0;
        int count = 0; 
        int ans = 0;   
        int prefix = 0; 

        for (int right = 0; right < nums.length; right++) {
            if (nums[right] % 2 != 0) {
                count++;
            }
            
            if (count > k) {
                while (nums[left] % 2 == 0) {
                    left++;
                }
                left++;
                count--;
                prefix = 0;
            }
            
            if (count == k) {
                while (nums[left + prefix] % 2 == 0) {
                    prefix++;
                }
                ans += (prefix + 1);
            }
        }
        return ans;
    }
}
```

## DAY 37
### Number of Substrings Containing All Three Characters
```java
class Solution {
    public int numberOfSubstrings(String s) {
        int lastSeen[] = {-1,-1,-1};
        int cnt = 0;
        for(int i = 0; i<s.length(); i++){
            lastSeen[s.charAt(i) - 'a'] = i;
            cnt += 1 + Math.min(lastSeen[0],Math.min(lastSeen[1],lastSeen[2]));
        }
        return cnt;
    }
}
```

## DAY 38
### Maximum Points you can obtain from Cards
```java
class Solution {
    public int maxScore(int[] arr, int k) {
        int lsum = 0, rsum = 0, maxsum = 0;
        int r = arr.length - 1;
        for(int i=0 ; i<k ; i++){
            lsum += arr[i];
        }
        maxsum = lsum;
        for(int i = k-1; i>=0; i--){
            lsum -= arr[i];
            rsum += arr[r];
            r--;
            maxsum = Math.max(maxsum, (lsum + rsum));
        }
        return maxsum;
    }
}
```
