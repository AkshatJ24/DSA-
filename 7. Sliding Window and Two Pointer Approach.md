## DAY 30
### Sliding Window Maximum
```java
import java.util.ArrayDeque;
import java.util.Deque;

class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        int n = nums.length;
        if (n == 0 || k <= 0) return new int[0];

        // If k > n, return the max of the entire array (or adjust per your spec)
        if (k > n) {
            int max = Integer.MIN_VALUE;
            for (int v : nums) max = Math.max(max, v);
            return new int[] { max };
        }

        int[] res = new int[n - k + 1];
        int ri = 0;
        Deque<Integer> dq = new ArrayDeque<>(); // stores indices

        for (int i = 0; i < n; i++) {
            // Remove indices that are out of the current window [i - k + 1, i]
            while (!dq.isEmpty() && dq.peekFirst() <= i - k) dq.pollFirst();

            // Maintain decreasing order by value in dq
            while (!dq.isEmpty() && nums[dq.peekLast()] < nums[i]) dq.pollLast();

            dq.offerLast(i);

            // Record the max once the first full window is formed
            if (i >= k - 1) res[ri++] = nums[dq.peekFirst()];
        }

        return res;
    }
}
```

## Day 32
### Length of Longest Substring without any Repeating Character
```java
class Solution {
    public int lengthOfLongestSubstring(String str) {
        if (str.length() == 0) {
            return 0;
        }
        int left = 0;
        int maxLen = 0;
        Set<Character> charSet = new HashSet<>();
        for (int right = 0; right < str.length(); right++) {
            char currentChar = str.charAt(right);
            while (charSet.contains(currentChar)) {
                charSet.remove(str.charAt(left));
                left++;
            }
            charSet.add(currentChar);
            maxLen = Math.max(maxLen, right - left + 1);
        }
        return maxLen;
    }
}
```

## Day 33
### Max Consecutive Ones III
```java
class Solution {
    public int longestOnes(int[] nums, int k) {
        int zeroCount = 0;
        int maxLength = 0;
        int left = 0;

        for (int right = 0; right < nums.length; right++) {
            if (nums[right] == 0) {
                zeroCount++;
            }
            while (zeroCount > k) {
                if (nums[left] == 0) {
                    zeroCount--;
                }
                left++;
            }
            maxLength = Math.max(maxLength, right - left + 1);
        }
        return maxLength;
    }
}
```
